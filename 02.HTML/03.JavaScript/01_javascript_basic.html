1. JavaScriptとは

JavaScriptは、ウェブページに動的な機能を追加するためのプログラミング言語です。ブラウザ上で実行され、ユーザーとのインタラクションを可能にします。

特徴：
  • ブラウザで直接実行可能
  • HTMLやCSSと組み合わせて使用
  • 動的なウェブページの作成が可能
  • イベント処理やユーザーインタラクションに対応
// 基本的なJavaScriptの書き方 console.log("Hello, JavaScript!"); alert("ようこそJavaScriptの世界へ！");

2. 変数と定数

変数は値を格納するための「箱」のようなものです。JavaScriptでは主にlet、const、varを使って変数を宣言します。

// 変数の宣言 let name = "田中太郎"; // 変更可能な変数 const age = 25; // 変更不可能な定数 var city = "東京"; // 古い書き方（推奨されない） // 値の変更 name = "佐藤花子"; // OK // age = 30; // エラー！constは変更不可

3. データ型

JavaScriptには様々なデータ型があります。主要なものを見てみましょう。

// 基本的なデータ型 let text = "文字列"; // string（文字列） let number = 42; // number（数値） let isTrue = true; // boolean（真偽値） let nothing = null; // null（空の値） let undefined_var; // undefined（未定義） // データ型の確認 console.log(typeof text); // "string" console.log(typeof number); // "number" console.log(typeof isTrue); // "boolean"

4. 演算子

演算子を使って計算や比較を行います。

// 算術演算子 let a = 10; let b = 3; console.log(a + b); // 13 (加算) console.log(a - b); // 7 (減算) console.log(a * b); // 30 (乗算) console.log(a / b); // 3.333... (除算) console.log(a % b); // 1 (余り) // 比較演算子 console.log(a > b); // true console.log(a === 10); // true (厳密等価) console.log(a != b); // true (不等価)

5. 条件分岐

if文を使って、条件に応じて異なる処理を実行できます。

let score = 85; if (score >= 90) { console.log("優秀です！"); } else if (score >= 70) { console.log("良好です"); } else if (score >= 60) { console.log("合格です"); } else { console.log("もう少し頑張りましょう"); } // 三項演算子 let result = score >= 60 ? "合格" : "不合格"; console.log(result);

🔄 成績判定システム

6. 繰り返し処理

同じ処理を何度も実行したい場合は、繰り返し文を使います。

// for文 for (let i = 1; i <= 5; i++) { console.log("回数: " + i); } // while文 let count = 1; while (count <= 3) { console.log("カウント: " + count); count++; } // 配列の繰り返し let fruits = ["りんご", "バナナ", "オレンジ"]; for (let fruit of fruits) { console.log(fruit); }

7. 関数

関数は再利用可能なコードの塊です。処理をまとめて、必要な時に呼び出すことができます。

// 関数の定義 function greet(name) { return "こんにちは、" + name + "さん！"; } // 関数の呼び出し let message = greet("田中"); console.log(message); // "こんにちは、田中さん！" // アロー関数（ES6の書き方） const add = (a, b) => { return a + b; }; // より短縮した書き方 const multiply = (a, b) => a * b; console.log(add(3, 5)); // 8 console.log(multiply(4, 6)); // 24

🔄 挨拶関数を使ってみよう

8. オブジェクトと配列

複数のデータをまとめて管理する方法を学びましょう。

// オブジェクト let person = { name: "山田太郎", age: 30, city: "大阪", greet: function() { return "私は" + this.name + "です"; } }; console.log(person.name); // "山田太郎" console.log(person.greet()); // "私は山田太郎です" // 配列 let colors = ["赤", "青", "緑", "黄"]; console.log(colors[0]); // "赤" console.log(colors.length); // 4 // 配列のメソッド colors.push("紫"); // 要素を追加 console.log(colors); // ["赤", "青", "緑", "黄", "紫"]

9. DOM操作

DOM（Document Object Model）を操作して、HTMLの内容を動的に変更できます。

// 要素の取得 let element = document.getElementById("myId"); let elements = document.getElementsByClassName("myClass"); let element2 = document.querySelector(".myClass"); // 内容の変更 element.textContent = "新しいテキスト"; element.innerHTML = "<strong>太字のテキスト</strong>"; // スタイルの変更 element.style.color = "red"; element.style.fontSize = "20px"; // クラスの操作 element.classList.add("newClass"); element.classList.remove("oldClass"); element.classList.toggle("activeClass");

10. イベント処理

ユーザーの操作（クリック、入力など）に反応する処理を作れます。

// イベントリスナーの追加 button.addEventListener('click', function() { console.log('ボタンがクリックされました！'); }); // より具体的な例 document.getElementById('myButton').addEventListener('click', function(event) { event.target.textContent = 'クリックされました！'; event.target.style.backgroundColor = 'lightblue'; }); // 入力イベント input.addEventListener('input', function(event) { console.log('入力値: ' + event.target.value); });
